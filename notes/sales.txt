const mongoose = require('mongoose');

const saleSchema = mongoose.Schema({
    _id: String,
    invoiceNumber: Number,
    date: Date,
    month: String,
    clientName: String,
    statement: String,
    materialCode: String,
    quantity: Number,
    sellingPrice: Number,
    costPrice: Number
})

const Sales = mongoose.model('sale', saleSchema);
exports.Sales = Sales
